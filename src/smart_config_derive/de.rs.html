11impl ConfigField {
12    /// Returns `Option<_>`.
13    fn deserialize_param(&self, index: usize) -> proc_macro2::TokenStream {
14        let name_span = self.name.span();
15        if self.attrs.nest {
16            if Self::is_option(&self.ty) {
17                quote_spanned! {name_span=>
18                    ctx.deserialize_nested_config_opt(#index)
19                }
20            } else {
21                let default_fn = wrap_in_option(self.default_fn());
22                quote_spanned! {name_span=>
23                    ctx.deserialize_nested_config(#index, #default_fn)
24                }
25            }
26        } else {
27            quote_spanned! {name_span=>
28                ctx.deserialize_param(#index)
29            }
30        }
31    }
32}

34impl ConfigEnumVariant {
35    fn matches(&self, rename_rule: Option<RenameRule>) -> proc_macro2::TokenStream {
36        let mut all_names = self.expected_variants(rename_rule);
37        let name = all_names.next().unwrap();
38        let name_span = self.name.span();
39        quote_spanned!(name_span=> #name #(| #all_names)*)
40    }
41}

43impl ConfigContainer {
44    fn process_fields<'a>(
45        fields: &'a [ConfigField],
46        param_index: &'a mut usize,
47        nested_index: &'a mut usize,
48    ) -> (proc_macro2::TokenStream, Vec<proc_macro2::TokenStream>) {
49        let mut init = proc_macro2::TokenStream::default();
50        let fields = fields.iter().enumerate().map(|(i, field)| {
51            let index;
52            if field.attrs.nest {
53                index = *nested_index;
54                *nested_index += 1;
55            } else {
56                index = *param_index;
57                *param_index += 1;
58            }
59
60            let name = &field.name;
61            let value = field.deserialize_param(index);
62            let local_var = Ident::new(&format!("__{i}"), name.span());
63            init.extend(quote_spanned! {name.span()=>
64                let #local_var = #value;
65            });
66            quote_spanned!(name.span()=> #name: #local_var?)
67        });
68        let fields = fields.collect();
69        (init, fields)
70    }
71
72    fn derive_deserialize_config(&self) -> proc_macro2::TokenStream {
73        let name = &self.name;
74
75        let mut param_index = 0;
76        let mut nested_index = 0;
77        let instance = match &self.fields {
78            ConfigContainerFields::Struct(fields) => {
79                let (init, fields) =
80                    Self::process_fields(fields, &mut param_index, &mut nested_index);
81                quote!({
82                    #init
83                    Self { #(#fields,)* }
84                })
85            }
86            ConfigContainerFields::Enum { variants, .. } => {
87                let match_hands = variants.iter().map(|variant| {
88                    let name = &variant.name;
89                    let matches = variant.matches(self.attrs.rename_all);
90                    let (init, variant_fields) =
91                        Self::process_fields(&variant.fields, &mut param_index, &mut nested_index);
92                    quote!(#matches => {
93                        #init
94                        Self::#name { #(#variant_fields,)* }
95                    })
96                });
97                let match_hands: Vec<_> = match_hands.collect();
98                let tag_expr = quote! {
99                    ctx.deserialize_param::<&'static str>(#param_index)
100                };
101
102                quote! {{
103                    match #tag_expr? {
104                        #(#match_hands)*
105                        _ => ::core::unreachable!(),
106                    }
107                }}
108            }
109        };
110
111        let cr = self.cr(Span::call_site());
112        quote! {
113            impl #cr::DeserializeConfig for #name {
114                #[allow(unused_mut)]
115                fn deserialize_config(
116                    mut ctx: #cr::de::DeserializeContext<'_>,
117                ) -> ::core::result::Result<Self, #cr::DeserializeConfigError> {
118                    ::core::result::Result::Ok(#instance)
119                }
120            }
121        }
122    }
123}

125pub(crate) fn impl_deserialize_config(input: TokenStream) -> TokenStream {
126    let input: DeriveInput = syn::parse(input).unwrap();
127    let trait_impl = match ConfigContainer::new(&input) {
128        Ok(trait_impl) => trait_impl,
129        Err(err) => return err.into_compile_error().into(),
130    };
131    trait_impl.derive_deserialize_config().into()
132}
